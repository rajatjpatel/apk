#!/bin/bash
# =============================================================================
# UNREGISTER VM FROM HIGH AVAILABILITY
# Removes a VM from Pacemaker management (VM keeps running, just not HA)
# =============================================================================
#
# Usage: unregister-vm-ha <vm-name>
#
# =============================================================================

set -euo pipefail

RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m'

log_info()  { echo -e "${GREEN}[INFO]${NC} $1"; }
log_warn()  { echo -e "${YELLOW}[WARN]${NC} $1"; }
log_error() { echo -e "${RED}[ERROR]${NC} $1"; }

VM_NAME="${1:-}"
VM_CONFIG_DIR="/var/lib/libvirt/images/vm-configs"

if [ -z "${VM_NAME}" ]; then
    echo "Usage: bash $0 <vm-name>"
    echo ""
    echo "Currently HA-registered VMs:"
    crm resource status 2>/dev/null | grep "VirtualDomain" || echo "  (none)"
    exit 1
fi

RESOURCE_NAME="vm-ha-${VM_NAME}"

# Check resource exists
if ! crm resource status "${RESOURCE_NAME}" &>/dev/null; then
    log_error "Resource '${RESOURCE_NAME}' not found in Pacemaker"
    exit 1
fi

log_info "Unregistering VM '${VM_NAME}' from HA..."

# Stop the resource first (VM will be stopped gracefully by Pacemaker)
log_warn "This will STOP the VM if it's running via Pacemaker."
read -p "Continue? (yes/no): " CONFIRM
if [ "${CONFIRM}" != "yes" ]; then
    log_info "Aborted."
    exit 0
fi

# Remove ordering constraints
log_info "Removing constraints..."
crm configure delete "order-fs-before-${VM_NAME}" 2>/dev/null || true
crm configure delete "order-libvirt-before-${VM_NAME}" 2>/dev/null || true
crm configure delete "loc-prefer-${VM_NAME}" 2>/dev/null || true

# Stop and remove the resource
log_info "Stopping and removing resource..."
crm resource stop "${RESOURCE_NAME}" 2>/dev/null || true
sleep 10
crm configure delete "${RESOURCE_NAME}" 2>/dev/null || true

log_info "VM '${VM_NAME}' removed from Pacemaker HA ✓"
log_info "VM XML config preserved at: ${VM_CONFIG_DIR}/${VM_NAME}.xml"
log_warn "To manually start the VM: virsh start ${VM_NAME}"
