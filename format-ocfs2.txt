#!/bin/bash
# =============================================================================
# OCFS2 Filesystem Format Script
# Run ONCE from HOSPM08PIXX01 after multipath WWNs are confirmed
# DESTRUCTIVE - Will erase all data on mpathb, mpathc and mpathd
# =============================================================================

set -euo pipefail

RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m'

log_info()  { echo -e "${GREEN}[INFO]${NC} $(date '+%Y-%m-%d %H:%M:%S') $1"; }
log_warn()  { echo -e "${YELLOW}[WARN]${NC} $(date '+%Y-%m-%d %H:%M:%S') $1"; }
log_error() { echo -e "${RED}[ERROR]${NC} $(date '+%Y-%m-%d %H:%M:%S') $1"; }

CLUSTER_NAME="ha-kvm-cluster"
# Using WWID-based paths directly (aliases mpathb/c/d require multipath.conf applied first)
MPATH_VM="/dev/mapper/360050768108184bc380000000000015a"
MPATH_PRDATA1="/dev/mapper/360050768108184bc380000000000015b"
MPATH_PRDATA2="/dev/mapper/360050768108184bc380000000000015c"
VM_LABEL="vm-storage"
PRDATA1_LABEL="PRDATA1"
PRDATA2_LABEL="PRDATA2"

# =============================================================================
# SAFETY CHECKS
# =============================================================================

echo ""
echo -e "${RED}╔══════════════════════════════════════════════════════════╗${NC}"
echo -e "${RED}║  WARNING: DESTRUCTIVE OPERATION                         ║${NC}"
echo -e "${RED}║  This will FORMAT and ERASE all data on:                ║${NC}"
echo -e "${RED}║    ${MPATH_VM}  (4TB  VM Storage)          ║${NC}"
echo -e "${RED}║    ${MPATH_PRDATA1}  (15TB PRDATA1)              ║${NC}"
echo -e "${RED}║    ${MPATH_PRDATA2}  (15TB PRDATA2)              ║${NC}"
echo -e "${RED}║  Run ONLY ONCE from HOSPM08PIXX01                       ║${NC}"
echo -e "${RED}╚══════════════════════════════════════════════════════════╝${NC}"
echo ""

# Hostname check
if [ "$(hostname -s)" != "HOSPM08PIXX01" ]; then
    log_error "Run only from HOSPM08PIXX01!"
    exit 1
fi

# Single confirmation
read -p "Type 'YES' to confirm formatting all 3 disks: " CONFIRM1
if [ "${CONFIRM1}" != "YES" ]; then
    log_error "Confirmation failed. Exiting."
    exit 1
fi

# Verify devices exist
if ! ls ${MPATH_VM} &>/dev/null; then
    log_error "Device ${MPATH_VM} not found! Check multipath configuration."
    exit 1
fi
if ! ls ${MPATH_PRDATA1} &>/dev/null; then
    log_error "Device ${MPATH_PRDATA1} not found! Check multipath configuration."
    exit 1
fi
if ! ls ${MPATH_PRDATA2} &>/dev/null; then
    log_error "Device ${MPATH_PRDATA2} not found! Check multipath configuration."
    exit 1
fi

# Install ocfs2-tools if not present
if ! command -v mkfs.ocfs2 &>/dev/null; then
    log_info "mkfs.ocfs2 not found - installing ocfs2-tools..."
    zypper install -y ocfs2-tools
fi
if ! command -v mkfs.ocfs2 &>/dev/null; then
    log_error "mkfs.ocfs2 still not found after install. Abort."
    exit 1
fi

# Load ocfs2 kernel module
modprobe ocfs2 2>/dev/null || true

log_info "Device sizes:"
lsblk ${MPATH_VM}      2>/dev/null || true
lsblk ${MPATH_PRDATA1} 2>/dev/null || true
lsblk ${MPATH_PRDATA2} 2>/dev/null || true

log_info "Formatting mpathb as OCFS2 (VM Storage - 4TB)..."
log_info "Command: mkfs.ocfs2 -L ${VM_LABEL} -N 2 --cluster-name ${CLUSTER_NAME}"

mkfs.ocfs2 \
    --label "${VM_LABEL}" \
    -N 2 \
    --cluster-name "${CLUSTER_NAME}" \
    --cluster-stack "pcmk" \
    --block-size 4096 \
    --cluster-size 1048576 \
    --fs-feature-level "max-compat" \
    --verbose \
    "${MPATH_VM}"

log_info "mpathb formatted ✓"

log_info "Formatting mpathc as OCFS2 (PRDATA1 - 15TB)..."
mkfs.ocfs2 \
    --label "${PRDATA1_LABEL}" \
    -N 2 \
    --cluster-name "${CLUSTER_NAME}" \
    --cluster-stack "pcmk" \
    --block-size 4096 \
    --cluster-size 1048576 \
    --fs-feature-level "max-compat" \
    --verbose \
    "${MPATH_PRDATA1}"

log_info "mpathc formatted ✓"

log_info "Formatting mpathd as OCFS2 (PRDATA2 - 15TB)..."
mkfs.ocfs2 \
    --label "${PRDATA2_LABEL}" \
    -N 2 \
    --cluster-name "${CLUSTER_NAME}" \
    --cluster-stack "pcmk" \
    --block-size 4096 \
    --cluster-size 1048576 \
    --fs-feature-level "max-compat" \
    --verbose \
    "${MPATH_PRDATA2}"

log_info "mpathd formatted ✓"

log_info "Verifying OCFS2 filesystems..."
tunefs.ocfs2 -q -l ${MPATH_VM}
tunefs.ocfs2 -q -l ${MPATH_PRDATA1}
tunefs.ocfs2 -q -l ${MPATH_PRDATA2}

log_info "=== OCFS2 Format Complete ==="
log_info "Now proceed with cluster resource configuration:"
log_info "  /root/configure-cluster-resources.sh"
